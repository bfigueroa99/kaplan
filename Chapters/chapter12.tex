\section{Reading Context}
\begin{readingbox}[title=Dialogue: Christmas in Summer?]
\textbf{Tom (UK):} I can't imagine Christmas without snow. It must be strange in Australia.\
\textbf{Sarah (Australia):} Not at all! For us, it's normal. Christmas \textbf{is celebrated} in summer.\
\textbf{Tom:} So what do you do? Do you eat roast turkey?\
\textbf{Sarah:} Sometimes, but often we have a barbecue on the beach. Seafood \textbf{is eaten} instead of heavy meat.\
\textbf{Tom:} That sounds different. In the UK, stockings \textbf{are hung} by the fireplace.\
\textbf{Sarah:} We do that too, but Santa Claus \textbf{is sometimes depicted} wearing shorts!
\end{readingbox}

\section{Key Concepts: Describing Traditions}

\begin{vocabbox}[title=Vocabulary]
\begin{itemize}
    \item \textbf{Custom:} A traditional way of behaving (e.g., taking off shoes). \trans{Costumbre}
    \item \textbf{Tradition:} A belief or behavior passed down within a group. \trans{Tradición}
    \item \textbf{Celebration:} A social gathering for a special event. \trans{Celebración}
    \item \textbf{Parade:} A public procession with music and costumes. \trans{Desfile}
    \item \textbf{Feast:} A large meal. \trans{Banquete}
    \item \textbf{To take place:} To happen. \trans{Tener lugar}
\end{itemize}
\end{vocabbox}

\section{Grammar Focus: The Passive Voice}

We often use the **Passive Voice** to describe traditions when the action is more important than who does it.

\begin{grammarbox}[title=Structure]
\textbf{Subject + Be + Past Participle (+ by Agent)}
\end{grammarbox}

\begin{grammarbox}[title=Examples]
\textbf{Present Passive (am/is/are + V3):}
\begin{itemize}
    \item Christmas \textbf{is celebrated} on December 25th. (Active: People celebrate Christmas...)
    \item Gifts \textbf{are exchanged} between friends.
    \item Special food \textbf{is eaten}.
\end{itemize}

\textbf{Past Passive (was/were + V3):}
\begin{itemize}
    \item The tradition \textbf{was started} in the 19th century.
    \item The poem \textbf{was written} by Clement Clarke Moore.
    \item The city \textbf{was decorated} with lights.
\end{itemize}
\end{grammarbox}

\section{Practice Exercises}

\subsection{Exercise 1: Active to Passive}
Rewrite the sentences in the Passive Voice.

\begin{enumerate}
    \item People celebrate Halloween in October. \
    $\rightarrow$ Halloween \underline{\hspace{6cm}}
    \item They eat turkey on Thanksgiving. \
    $\rightarrow$ Turkey \underline{\hspace{6cm}}
    \item Coca-Cola popularized the red suit. \
    $\rightarrow$ The red suit \underline{\hspace{6cm}}
    \item Someone wrote this song in 1950. \
    $\rightarrow$ This song \underline{\hspace{6cm}}
\end{enumerate}

\section{Key Takeaways}
\begin{itemize}
    \item Use the \textbf{Passive Voice} (is/was + past participle) to focus on the action/event.
    \item Vocabulary: Custom, Tradition, Parade, Feast.
    \item Traditions vary greatly around the world (e.g., KFC in Japan for Christmas).
\end{itemize}
