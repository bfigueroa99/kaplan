\section{Grammar Focus: Present Perfect Formation}

The Present Perfect connects the past with the present. It describes actions that happened at an unspecified time or that have relevance now.

\begin{grammarbox}[title=Present Perfect Structure]
\textbf{Structure:} Subject + \textbf{have/has} + \textbf{past participle (V3)}

\textbf{Affirmative:}
\begin{itemize}
    \item I/You/We/They + \textbf{have} + V3 (I've worked)
    \item He/She/It + \textbf{has} + V3 (She's worked)
\end{itemize}

\textbf{Negative:}
\begin{itemize}
    \item I/You/We/They + \textbf{haven't} + V3
    \item He/She/It + \textbf{hasn't} + V3
\end{itemize}

\textbf{Questions:}
\begin{itemize}
    \item \textbf{Have} + I/you/we/they + V3?
    \item \textbf{Has} + he/she/it + V3?
\end{itemize}

\textbf{Examples:}
\begin{itemize}
    \item I \textbf{have visited} London. \trans{He visitado Londres}
    \item She \textbf{has finished} her work. \trans{Ella ha terminado su trabajo}
    \item They \textbf{haven't seen} the movie. \trans{No han visto la película}
    \item \textbf{Have} you \textbf{been} to Paris? \trans{¿Has estado en París?}
\end{itemize}
\end{grammarbox}

\section{When to Use Present Perfect}

\begin{grammarbox}[title=Use 1: Life Experiences (Unspecified Time)]
When you don't say exactly WHEN something happened.

\textbf{Examples:}
\begin{itemize}
    \item I \textbf{have been} to France. \trans{He estado en Francia} (sometime in my life)
    \item She \textbf{has seen} that movie. \trans{Ella ha visto esa película}
    \item \textbf{Have} you \textbf{ever tried} sushi? \trans{¿Alguna vez has probado sushi?}
\end{itemize}

\textbf{Common with:} ever, never, before
\end{grammarbox}

\begin{grammarbox}[title=Use 2: Recent Actions with Relevance Now]
Actions that happened recently and are important now.

\textbf{Examples:}
\begin{itemize}
    \item I\textbf{'ve just finished} my homework. \trans{Acabo de terminar} (now I'm free)
    \item She\textbf{'s just arrived}. \trans{Acaba de llegar} (she's here now)
    \item They\textbf{'ve already eaten}. \trans{Ya han comido} (they're not hungry)
\end{itemize}

\textbf{Common with:} just, already, yet
\end{grammarbox}

\begin{grammarbox}[title=Use 3: Actions Continuing from Past to Present]
Actions that started in the past and continue now.

\textbf{Examples:}
\begin{itemize}
    \item I\textbf{'ve lived} here \textbf{for} 5 years. \trans{He vivido aquí por 5 años} (still living here)
    \item She\textbf{'s worked} there \textbf{since} 2020. \trans{Ha trabajado desde 2020} (still works there)
\end{itemize}

\textbf{Common with:} for, since, How long...?
\end{grammarbox}

\section{Present Perfect vs. Past Simple}

\begin{table}[h]
\centering
\begin{tabular}{|p{5.5cm}|p{5.5cm}|}
\hline
\textbf{Present Perfect} & \textbf{Past Simple} \\
\hline
\multicolumn{2}{|c|}{\textbf{Key Difference}} \\
\hline
Time NOT specified (or not important) & Specific time IS mentioned \\
\hline
Connection to present & Finished and disconnected from now \\
\hline
\textbf{Examples:} & \textbf{Examples:} \\
\hline
I\textbf{'ve visited} Paris. (sometime) & I \textbf{visited} Paris \textbf{last year}. \\
\hline
She\textbf{'s lived} here for 3 years. (still here) & She \textbf{lived} here in 2015. (not anymore) \\
\hline
\textbf{Have} you \textbf{seen} the movie? (ever) & \textbf{Did} you \textbf{see} it \textbf{yesterday}? \\
\hline
He\textbf{'s just left}. (just now - he's gone) & He \textbf{left} \textbf{at 3 PM}. (specific time) \\
\hline
\end{tabular}
\caption{Present Perfect vs. Past Simple}
\end{table}

\begin{examplebox}[title=When to Use Each Tense]
\textbf{Cannot use Present Perfect with specific past time:}
\begin{itemize}
    \item \textcolor{red}{I have visited Paris last year.} \textbf{X}
    \item \textcolor{green!60!black}{I visited Paris last year.} \checkmark
\end{itemize}

\textbf{Time expressions that require Past Simple:}
\begin{itemize}
    \item yesterday, last week/month/year, ago, in 2020, when I was young
\end{itemize}

\textbf{Time expressions with Present Perfect:}
\begin{itemize}
    \item ever, never, just, already, yet, recently, so far, up to now
\end{itemize}
\end{examplebox}

\section{Key Adverbs: Ever, Never, Just, Already, Yet}

\begin{vocabbox}[title=Important Adverbs with Present Perfect]
\textbf{EVER} (alguna vez) - questions
\begin{itemize}
    \item \textbf{Have} you \textbf{ever been} to China? \trans{¿Alguna vez has estado en China?}
\end{itemize}

\textbf{NEVER} (nunca) - negative
\begin{itemize}
    \item I\textbf{'ve never seen} snow. \trans{Nunca he visto nieve}
\end{itemize}

\textbf{JUST} (recién/acabar de) - recent action
\begin{itemize}
    \item She\textbf{'s just left}. \trans{Acaba de irse}
\end{itemize}

\textbf{ALREADY} (ya) - affirmative
\begin{itemize}
    \item I\textbf{'ve already finished}. \trans{Ya he terminado}
\end{itemize}

\textbf{YET} (todavía/ya) - questions and negatives
\begin{itemize}
    \item \textbf{Have} you finished \textbf{yet}? \trans{¿Ya terminaste?}
    \item I \textbf{haven't finished yet}. \trans{Todavía no he terminado}
\end{itemize}
\end{vocabbox}

\section{FOR vs. SINCE}

\begin{grammarbox}[title=FOR = Duration (how long)]
Use \textbf{for} + period of time

\textbf{Examples:}
\begin{itemize}
    \item for 2 hours, for 3 weeks, for 5 years, for a long time
    \item I\textbf{'ve worked} here \textbf{for} 10 years. \trans{He trabajado aquí por 10 años}
\end{itemize}
\end{grammarbox}

\begin{grammarbox}[title=SINCE = Starting Point (when it started)]
Use \textbf{since} + specific point in time

\textbf{Examples:}
\begin{itemize}
    \item since Monday, since 2020, since January, since I was a child
    \item She\textbf{'s lived} here \textbf{since} 2018. \trans{Ha vivido aquí desde 2018}
\end{itemize}
\end{grammarbox}

\section{Key Takeaways}
\begin{itemize}
    \item \textbf{Present Perfect:} have/has + past participle (V3)
    \item Use for: life experiences (unspecified time), recent actions, continuing situations
    \item \textbf{Cannot use} with specific past time (yesterday, last year, ago)
    \item \textbf{FOR} = duration (for 5 years), \textbf{SINCE} = starting point (since 2020)
    \item \textbf{Ever/never} for experiences, \textbf{just/already/yet} for recent actions
    \item Past Simple = finished action with specific time; Present Perfect = connection to now
\end{itemize}
