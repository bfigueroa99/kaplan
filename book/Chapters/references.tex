\section{IELTS Preparation Guide}

\subsection{Listening \textipa{/"lIs@nIN/} (Audio + Answers)}

\textbf{Full Tests}
\begin{itemize}
    \item \href{https://ielts-up.com/listening/ielts-listening-practice.html}{IELTS-up Listening Practice}
    \item \href{https://ieltsonlinetests.com/ielts-listening-tests}{IELTS Online Tests – Listening}
    \item \href{https://takeielts.britishcouncil.org/take-ielts/prepare/free-ielts-english-practice-tests/listening}{British Council – Sample Listening Tests}
    \item \href{https://ieltstrainingonline.com/practice-cam-20-listening-test-01-with-answer-and-audioscripts/}{IELTS Training Online – Cam 20 Listening Test 01}
\end{itemize}

\textbf{YouTube Practice}
\begin{itemize}
    \item \href{https://www.youtube.com/results?search_query=ielts+listening+practice+test+with+answers}{IELTS Listening Practice Tests (YouTube Search)}
\end{itemize}

\noindent\textbf{Tips:}
\begin{itemize}
    \item Simulate real conditions: 30 min audio without pausing.
    \item 10 min only to transfer answers.
    \item Correct using the script provided.
\end{itemize}

\subsection{Reading \textipa{/"ri:dIN/} (With Answers)}

\textbf{General Training Reading Tests}
\begin{itemize}
    \item \href{https://ieltsonlinetests.com/ielts-reading-tests}{IELTS Online Tests – Reading}
    \item \href{https://www.ieltsfreetests.com/reading}{IELTS Free Tests – Reading (GT)}
    \item \href{https://takeielts.britishcouncil.org/take-ielts/prepare/free-ielts-english-practice-tests/reading}{British Council – Sample Reading}
\end{itemize}

\noindent\textbf{Tips:}
\begin{itemize}
    \item Timer set to 60 minutes.
    \item Cover 3 complete sections.
    \item Always review every wrong answer.
\end{itemize}

\subsection{Writing \textipa{/"raItIN/} (Tasks + Examples)}

\textbf{Tasks and Model Answers}
\begin{itemize}
    \item \href{https://www.ieltsfreetests.com/writing}{IELTS Free Tests – Writing}
    \item \href{https://ieltsliz.com/ielts-writing-task-1-general/}{IELTS Liz – Task 1 General}
    \item \href{https://ieltsliz.com/ielts-writing-task-2/}{IELTS Liz – Task 2}
    \item \href{https://takeielts.britishcouncil.org/take-ielts/prepare/free-ielts-english-practice-tests/writing}{British Council – Writing Samples}
\end{itemize}

\noindent\textbf{Tips:}
\begin{itemize}
    \item Task 1: 20 min (150+ words).
    \item Task 2: 40 min (250+ words).
    \item Use model answers to compare structure and vocabulary.
\end{itemize}

\subsection{Speaking \textipa{/"spi:kIN/} (Cards + Examples)}

\textbf{Topics and Model Answers}
\begin{itemize}
    \item \href{https://www.ieltsfreetests.com/speaking}{IELTS Free Tests – Speaking}
    \item \href{https://takeielts.britishcouncil.org/take-ielts/prepare/free-ielts-english-practice-tests/speaking}{British Council – Speaking Sample}
    \item \href{https://ieltsliz.com/ielts-speaking-topics-and-questions/}{IELTS Liz – Speaking Topics \& Answers}
    \item \href{https://ieltsliz.com/ielts-speaking-part-2-topics/}{IELTS Liz – Speaking Part 2 Topics}
\end{itemize}

\noindent\textbf{Tips:}
\begin{itemize}
    \item Record yourself for 20–30 min daily.
    \item Practice Part 1, Part 2, and Part 3.
    \item Do not memorize — focus on ideas and connectors.
\end{itemize}
