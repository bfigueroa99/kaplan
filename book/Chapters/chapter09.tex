\chapter{Prepositions of Place and Time}

\section{Lesson Objectives}
In this chapter, you will learn:
\begin{itemize}
    \item How to use prepositions IN, AT, and ON for place
    \item How to use prepositions IN, AT, and ON for time
    \item Common prepositional phrases
    \item Prepositions with transportation
    \item Common mistakes and how to avoid them
\end{itemize}

\section{Reading Context}
\begin{readingbox}[title=Dialogue: Meeting Up]
\textbf{Anna:} Where shall we meet?\\
\textbf{Tom:} How about \textbf{at} the coffee shop \textbf{in} the city center?\\
\textbf{Anna:} Which one? There are many coffee shops \textbf{in} that area.\\
\textbf{Tom:} The one \textbf{on} Baker Street, next to the bookstore.\\
\textbf{Anna:} Oh, I know it! What time? I finish work \textbf{at} 5 PM.\\
\textbf{Tom:} Perfect. Let's meet \textbf{at} 5:30 then. Are you coming by car or \textbf{on} the bus?\\
\textbf{Anna:} I'll come \textbf{on} the bus. I'll be there \textbf{on} time!\\
\textbf{Tom:} Great! See you \textbf{on} Friday.
\end{readingbox}

\section{Prepositions of Place}

\subsection{IN - Inside an Area or Space}

\begin{grammarbox}[title=Preposition IN - Place]
\textbf{Use IN for:}
\begin{itemize}
    \item \textbf{Countries and cities:} in Spain, in London, in Tokyo
    \item \textbf{Rooms and buildings:} in the kitchen, in the office, in a hotel
    \item \textbf{Enclosed spaces:} in a box, in a drawer, in my bag
    \item \textbf{Large areas:} in the park, in the garden, in the street (UK)
    \item \textbf{Vehicles (enclosed):} in a car, in a taxi, in a helicopter
\end{itemize}

\textbf{Examples:}
\begin{itemize}
    \item I live \textbf{in} London. \trans{Vivo en Londres}
    \item The keys are \textbf{in} the drawer. \trans{Las llaves están en el cajón}
    \item She's \textbf{in} the office. \trans{Está en la oficina}
    \item Get \textbf{in} the car. \trans{Sube al coche}
\end{itemize}
\end{grammarbox}

\subsection{AT - A Specific Point or Location}

\begin{grammarbox}[title=Preposition AT - Place]
\textbf{Use AT for:}
\begin{itemize}
    \item \textbf{Specific points:} at the bus stop, at the entrance, at the door
    \item \textbf{Events and gatherings:} at a party, at a meeting, at a concert
    \item \textbf{Addresses (exact location):} at 25 Oxford Street
    \item \textbf{Fixed expressions:} at home, at work, at school, at university
\end{itemize}

\textbf{Examples:}
\begin{itemize}
    \item She is \textbf{at} the bus stop. \trans{Ella está en la parada de autobús}
    \item We met \textbf{at} the entrance. \trans{Nos encontramos en la entrada}
    \item I'll see you \textbf{at} the party. \trans{Te veré en la fiesta}
    \item He's \textbf{at} home. \trans{Está en casa}
\end{itemize}
\end{grammarbox}

\subsection{ON - On a Surface}

\begin{grammarbox}[title=Preposition ON - Place]
\textbf{Use ON for:}
\begin{itemize}
    \item \textbf{Surfaces:} on the table, on the wall, on the floor
    \item \textbf{Streets (US):} on Baker Street, on Fifth Avenue
    \item \textbf{Floors of buildings:} on the second floor
    \item \textbf{Public transportation (where you can walk):} on the bus, on the train, on a plane
    \item \textbf{Motorcycles and bicycles:} on a motorcycle, on a bike
\end{itemize}

\textbf{Examples:}
\begin{itemize}
    \item The book is \textbf{on} the table. \trans{El libro está sobre la mesa}
    \item The picture is \textbf{on} the wall. \trans{El cuadro está en la pared}
    \item I live \textbf{on} Baker Street. (US) \trans{Vivo en la calle Baker}
    \item He is \textbf{on} the train. \trans{Él está en el tren}
\end{itemize}
\end{grammarbox}

\section{Prepositions of Time}

\subsection{IN - Months, Years, Centuries, Long Periods}

\begin{grammarbox}[title=Preposition IN - Time]
\textbf{Use IN for:}
\begin{itemize}
    \item \textbf{Months:} in January, in December
    \item \textbf{Years:} in 2023, in 1990
    \item \textbf{Decades/Centuries:} in the 1990s, in the 21st century
    \item \textbf{Seasons:} in summer, in winter
    \item \textbf{Parts of day:} in the morning, in the afternoon, in the evening
    \item \textbf{Future time expressions:} in 10 minutes, in 2 hours, in a week
\end{itemize}

\textbf{Examples:}
\begin{itemize}
    \item I was born \textbf{in} 1995. \trans{Nací en 1995}
    \item We go skiing \textbf{in} winter. \trans{Vamos a esquiar en invierno}
    \item I wake up \textbf{in} the morning. \trans{Me despierto por la mañana}
    \item I'll call you \textbf{in} 10 minutes. \trans{Te llamo en 10 minutos}
\end{itemize}

\textbf{Exception:} at night (NOT in the night)
\end{grammarbox}

\subsection{AT - Specific Times}

\begin{grammarbox}[title=Preposition AT - Time]
\textbf{Use AT for:}
\begin{itemize}
    \item \textbf{Clock times:} at 3 PM, at 9:30, at midnight, at noon
    \item \textbf{Meal times:} at breakfast, at lunch, at dinner
    \item \textbf{Festivals:} at Christmas, at Easter
    \item \textbf{Fixed expressions:} at night, at the weekend (UK), at the moment
\end{itemize}

\textbf{Examples:}
\begin{itemize}
    \item The meeting is \textbf{at} 3 PM. \trans{La reunión es a las 3 PM}
    \item I eat \textbf{at} lunchtime. \trans{Como a la hora del almuerzo}
    \item We visit family \textbf{at} Christmas. \trans{Visitamos familia en Navidad}
    \item I'm busy \textbf{at} the moment. \trans{Estoy ocupado en este momento}
\end{itemize}
\end{grammarbox}

\subsection{ON - Days and Dates}

\begin{grammarbox}[title=Preposition ON - Time]
\textbf{Use ON for:}
\begin{itemize}
    \item \textbf{Days of the week:} on Monday, on Friday, on weekdays
    \item \textbf{Specific dates:} on December 25th, on my birthday
    \item \textbf{Special days:} on New Year's Day, on Valentine's Day
\end{itemize}

\textbf{Examples:}
\begin{itemize}
    \item I'll see you \textbf{on} Monday. \trans{Te veo el lunes}
    \item The meeting is \textbf{on} December 15th. \trans{La reunión es el 15 de diciembre}
    \item We don't work \textbf{on} weekends. \trans{No trabajamos los fines de semana}
    \item Happy birthday! Was it \textbf{on} Friday? \trans{¿Fue el viernes?}
\end{itemize}
\end{grammarbox}

\section{Transportation Prepositions}

\begin{table}[h]
\centering
\begin{tabular}{|l|l|}
\hline
\textbf{Use IN} & \textbf{Use ON} \\
\hline
in a car & on a bus \\
in a taxi & on a train \\
in a helicopter & on a plane \\
in a small boat & on a ship \\
& on a motorcycle \\
& on a bicycle \\
\hline
\end{tabular}
\caption{Prepositions with transportation}
\end{table}

\textbf{Rule:} Use \textbf{IN} for enclosed vehicles where you sit. Use \textbf{ON} for larger vehicles where you can walk, or vehicles you ride.

\section{Common Mistakes}

\begin{examplebox}[title=Typical Errors]
\begin{itemize}
    \item \textcolor{red}{I live \textbf{at} London.} \textbf{X}
    \item \textcolor{green!60!black}{I live \textbf{in} London.} \checkmark
    \vspace{0.3cm}
    \item \textcolor{red}{The meeting is \textbf{in} Monday.} \textbf{X}
    \item \textcolor{green!60!black}{The meeting is \textbf{on} Monday.} \checkmark
    \vspace{0.3cm}
    \item \textcolor{red}{I wake up \textbf{at} the morning.} \textbf{X}
    \item \textcolor{green!60!black}{I wake up \textbf{in} the morning.} \checkmark
    \vspace{0.3cm}
    \item \textcolor{red}{See you \textbf{in} 5 PM.} \textbf{X}
    \item \textcolor{green!60!black}{See you \textbf{at} 5 PM.} \checkmark
    \vspace{0.3cm}
    \item \textcolor{red}{I'm \textbf{in} the bus.} \textbf{X}
    \item \textcolor{green!60!black}{I'm \textbf{on} the bus.} \checkmark
\end{itemize}
\end{examplebox}

\section{Other Important Prepositions}

\begin{vocabbox}[title=Additional Prepositions]
\begin{itemize}
    \item \textbf{NEXT TO / BESIDE:} next to the bank \trans{al lado de}
    \item \textbf{BETWEEN:} between the bank and the post office \trans{entre}
    \item \textbf{OPPOSITE:} opposite the station \trans{frente a}
    \item \textbf{BEHIND:} behind the house \trans{detrás de}
    \item \textbf{IN FRONT OF:} in front of the building \trans{delante de}
    \item \textbf{ABOVE / OVER:} above/over the door \trans{encima de}
    \item \textbf{BELOW / UNDER:} below/under the table \trans{debajo de}
    \item \textbf{NEAR:} near the station \trans{cerca de}
\end{itemize}
\end{vocabbox}

\section{Key Takeaways}
\begin{itemize}
    \item \textbf{IN:} cities, countries, rooms, months, years, mornings/afternoons/evenings
    \item \textbf{AT:} specific points, addresses, clock times, night, weekend
    \item \textbf{ON:} surfaces, streets (US), days, dates
    \item \textbf{Transportation:} IN enclosed vehicles (car, taxi), ON larger vehicles (bus, train, plane)
    \item Remember fixed expressions: at home, at work, at school, at night, in the morning
\end{itemize}
