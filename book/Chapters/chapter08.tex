\section{Reading Context}
\begin{readingbox}[title=Dialogue: Getting to Know Someone]
\textbf{Anna:} Hi! \textbf{What's} your name?\\
\textbf{Ben:} I'm Ben. \textbf{What about} you?\\
\textbf{Anna:} I'm Anna. \textbf{Where are} you \textbf{from}?\\
\textbf{Ben:} I'm from Canada. \textbf{How long have} you \textbf{been} in London?\\
\textbf{Anna:} For two years. \textbf{What do} you \textbf{do}?\\
\textbf{Ben:} I'm a teacher. \textbf{Do} you \textbf{like} living here?\\
\textbf{Anna:} Yes, I do. The weather's not great though, \textbf{is it}?\\
\textbf{Ben:} No, it isn't. Could you tell me \textbf{where the nearest coffee shop is}?\\
\textbf{Anna:} Sure! It's just around the corner.
\end{readingbox}

\begin{britishbox}
\textbf{British Question Etiquette:} British people often use \textbf{indirect questions} to be polite: \textit{"Could you tell me where...?"} instead of \textit{"Where is...?"} Question tags are very common: \textit{"It's cold today, isn't it?"} They're used to invite agreement or start conversation. When asking for help, British speakers typically say: \textit{"Excuse me, I wonder if you could help me?"} or \textit{"Sorry to bother you, but do you know...?"} Using \textbf{sorry} before questions is common British politeness, even when you've done nothing wrong!
\end{britishbox}

\section{Grammar Focus: Yes/No Questions}

These questions can be answered with "yes" or "no".

\begin{grammarbox}[title=Yes/No Question Formation]
\textbf{Structure:} Auxiliary/Modal + Subject + Verb?

\textbf{Present Simple:}
\begin{itemize}
    \item \textbf{Do} you like coffee? \trans{¿Te gusta el café?}
    \item \textbf{Does} she work here? \trans{¿Trabaja ella aquí?}
\end{itemize}

\textbf{Present Continuous:}
\begin{itemize}
    \item \textbf{Are} you studying? \trans{¿Estás estudiando?}
    \item \textbf{Is} he coming? \trans{¿Viene él?}
\end{itemize}

\textbf{Past Simple:}
\begin{itemize}
    \item \textbf{Did} you go? \trans{¿Fuiste?}
    \item \textbf{Did} she call? \trans{¿Llamó ella?}
\end{itemize}

\textbf{Present Perfect:}
\begin{itemize}
    \item \textbf{Have} you finished? \trans{¿Has terminado?}
    \item \textbf{Has} he arrived? \trans{¿Ha llegado?}
\end{itemize}

\textbf{With Modal Verbs:}
\begin{itemize}
    \item \textbf{Can} you swim? \trans{¿Sabes nadar?}
    \item \textbf{Will} you help me? \trans{¿Me ayudarás?}
\end{itemize}
\end{grammarbox}

\section{WH-Questions (Information Questions)}

These questions ask for specific information using question words.

\begin{vocabbox}[title=Question Words]
\begin{itemize}
    \item \textbf{What} - asks about things: What is your name?
    \item \textbf{Who} - asks about people: Who is that?
    \item \textbf{Where} - asks about places: Where do you live?
    \item \textbf{When} - asks about time: When did you arrive?
    \item \textbf{Why} - asks about reasons: Why are you late?
    \item \textbf{Which} - asks about choices: Which do you prefer?
    \item \textbf{How} - asks about manner: How did you do it?
    \item \textbf{Whose} - asks about possession: Whose book is this?
\end{itemize}
\end{vocabbox}

\begin{grammarbox}[title=WH-Question Structure]
\textbf{Structure:} WH-word + Auxiliary + Subject + Verb?

\textbf{Examples:}
\begin{itemize}
    \item \textbf{What} do you do? \trans{¿A qué te dedicas?}
    \item \textbf{Where} does she live? \trans{¿Dónde vive ella?}
    \item \textbf{When} did they arrive? \trans{¿Cuándo llegaron?}
    \item \textbf{Why} are you crying? \trans{¿Por qué lloras?}
    \item \textbf{How} can I help you? \trans{¿Cómo puedo ayudarte?}
\end{itemize}
\end{grammarbox}

\section{Subject Questions vs. Object Questions}

\begin{grammarbox}[title=Subject Questions]
When asking about the \textbf{subject}, NO auxiliary verb is needed.

\textbf{Structure:} WH-word + Verb + Object?

\textbf{Examples:}
\begin{itemize}
    \item \textbf{Who} called you? (NOT: Who did call you?)
    \item \textbf{What} happened? (NOT: What did happen?)
    \item \textbf{Who} broke the window?
\end{itemize}

\trans{¿Quién te llamó? / ¿Qué pasó? / ¿Quién rompió la ventana?}
\end{grammarbox}

\begin{grammarbox}[title=Object Questions]
When asking about the \textbf{object}, use auxiliary verb.

\textbf{Structure:} WH-word + Auxiliary + Subject + Verb?

\textbf{Examples:}
\begin{itemize}
    \item \textbf{Who} did you call? (you = subject; who = object)
    \item \textbf{What} did she say?
    \item \textbf{Where} did they go?
\end{itemize}

\trans{¿A quién llamaste? / ¿Qué dijo ella? / ¿Adónde fueron?}
\end{grammarbox}

\section{Question Tags}

Question tags are short questions at the end of statements, used to confirm information or invite agreement.

\begin{grammarbox}[title=Question Tag Rules]
\textbf{Rule 1:} Positive statement $\rightarrow$ negative tag
\begin{itemize}
    \item You like coffee, \textbf{don't you}? \trans{Te gusta el café, ¿verdad?}
    \item She's French, \textbf{isn't she}? \trans{Ella es francesa, ¿no?}
\end{itemize}

\textbf{Rule 2:} Negative statement $\rightarrow$ positive tag
\begin{itemize}
    \item You don't smoke, \textbf{do you}? \trans{No fumas, ¿verdad?}
    \item He isn't coming, \textbf{is he}? \trans{No viene, ¿no?}
\end{itemize}

\textbf{Common Tags:}
\begin{itemize}
    \item is/isn't, are/aren't, was/wasn't, were/weren't
    \item do/don't, does/doesn't, did/didn't
    \item can/can't, will/won't, have/haven't
\end{itemize}
\end{grammarbox}

\section{Indirect Questions (Polite Questions)}

Indirect questions are more polite and formal.

\begin{grammarbox}[title=Indirect Question Structure]
\textbf{Direct:} Where is the station?

\textbf{Indirect:} Could you tell me \textbf{where the station is}?

\textbf{Notice:} In indirect questions, use STATEMENT word order (no auxiliary before subject).

\textbf{Common Introductions:}
\begin{itemize}
    \item Could you tell me...?
    \item Do you know...?
    \item I wonder...
    \item Would you mind telling me...?
    \item Can you explain...?
\end{itemize}

\textbf{More Examples:}
\begin{itemize}
    \item Direct: What time does the train leave?
    \item Indirect: Do you know what time the train leaves?
    \vspace{0.3cm}
    \item Direct: Where can I buy tickets?
    \item Indirect: Could you tell me where I can buy tickets?
\end{itemize}
\end{grammarbox}

\section{Key Takeaways}
\begin{itemize}
    \item \textbf{Yes/No questions:} Auxiliary + Subject + Verb?
    \item \textbf{WH-questions:} WH-word + Auxiliary + Subject + Verb?
    \item \textbf{Subject questions:} No auxiliary needed (Who called?)
    \item \textbf{Question tags:} Positive statement $\rightarrow$ negative tag (and vice versa)
    \item \textbf{Indirect questions:} Use statement word order for politeness
    \item Use \textbf{How much} with uncountable nouns, \textbf{How many} with countable
\end{itemize}
