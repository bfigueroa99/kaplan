\subsection{Conditionals}

\begin{itemize}
    \item Zero Conditional: General truths
    \begin{itemize}
        \item If + present simple, present simple
        \item Example: If you heat water, it boils.
    \end{itemize}
    \item First Conditional: Real future possibilities
    \begin{itemize}
        \item If + present simple, will + base verb
        \item Example: If it rains, we will stay inside.
    \end{itemize}
    \item Second Conditional: Unreal present/future situations
    \begin{itemize}
        \item If + past simple, would + base verb
        \item Example: If I were rich, I would travel the world.
    \end{itemize}
    \item Third Conditional: Unreal past situations
    \begin{itemize}
        \item If + past perfect, would have + past participle
        \item Example: If I had studied harder, I would have passed the exam.
    \end{itemize}
\end{itemize}
