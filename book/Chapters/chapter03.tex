\section{Grammar Focus: Indefinite Articles (A / AN)}

Use \textbf{a} or \textbf{an} before \textbf{singular countable nouns} when talking about something \textbf{for the first time} or \textbf{in general}.

\begin{grammarbox}[title=A vs. AN]
\textbf{Use A:} Before consonant \textbf{sounds}
\begin{itemize}
    \item \textbf{a} book \trans{un libro}
    \item \textbf{a} car \trans{un coche}
    \item \textbf{a} university (sounds like "yuniversity")
    \item \textbf{a} European country (sounds like "yuropean")
\end{itemize}

\textbf{Use AN:} Before vowel \textbf{sounds} (a, e, i, o, u)
\begin{itemize}
    \item \textbf{an} apple \trans{una manzana}
    \item \textbf{an} orange \trans{una naranja}
    \item \textbf{an} hour (h is silent)
    \item \textbf{an} honest person (h is silent)
\end{itemize}

\textbf{Important:} It depends on the SOUND, not the letter!
\end{grammarbox}

\section{Grammar Focus: Definite Article (THE)}

Use \textbf{the} when both speaker and listener know \textbf{which specific thing} you're talking about.

\begin{grammarbox}[title=When to Use THE]
\textbf{1. Second mention:}
\begin{itemize}
    \item I saw \textbf{a} dog. \textbf{The} dog was big. \trans{Vi un perro. El perro era grande}
\end{itemize}

\textbf{2. Unique things:}
\begin{itemize}
    \item \textbf{the} sun, \textbf{the} moon, \textbf{the} sky, \textbf{the} world
\end{itemize}

\textbf{3. Specific items (context makes it clear):}
\begin{itemize}
    \item Close \textbf{the} door. (we know which door)
    \item I'm going to \textbf{the} supermarket. (our local one)
\end{itemize}

\textbf{4. Superlatives:}
\begin{itemize}
    \item \textbf{the} best, \textbf{the} biggest, \textbf{the} most expensive
\end{itemize}

\textbf{5. Ordinal numbers:}
\begin{itemize}
    \item \textbf{the} first, \textbf{the} second, \textbf{the} last
\end{itemize}

\textbf{6. Geographical features:}
\begin{itemize}
    \item \textbf{the} River Thames, \textbf{the} Pacific Ocean, \textbf{the} Alps
\end{itemize}
\end{grammarbox}

\section{When NOT to Use Articles (Zero Article)}

\begin{grammarbox}[title=No Article Needed]
\textbf{1. General plural nouns:}
\begin{itemize}
    \item I like \textbf{-} apples. (apples in general)
    \item \textbf{-} Dogs are friendly. (dogs in general)
\end{itemize}

\textbf{2. Uncountable nouns (general):}
\begin{itemize}
    \item I drink \textbf{-} coffee. \trans{Tomo cafÃ©}
    \item \textbf{-} Water is important. \trans{El agua es importante}
\end{itemize}

\textbf{3. Most countries and cities:}
\begin{itemize}
    \item \textbf{-} Spain, \textbf{-} London, \textbf{-} France
    \item \textbf{BUT:} the UK, the USA, the Netherlands
\end{itemize}

\textbf{4. Languages:}
\begin{itemize}
    \item I speak \textbf{-} English and \textbf{-} Spanish.
\end{itemize}

\textbf{5. Meals (general):}
\begin{itemize}
    \item I have \textbf{-} breakfast at 8 AM.
    \item \textbf{BUT:} The breakfast was delicious. (specific)
\end{itemize}

\textbf{6. With possessives:}
\begin{itemize}
    \item This is \textbf{my} book. (NOT: the my book)
\end{itemize}
\end{grammarbox}

\section{Countable vs. Uncountable Nouns}

\begin{table}[h]
\centering
\begin{tabular}{|p{5.5cm}|p{5.5cm}|}
\hline
\textbf{Countable (Can count)} & \textbf{Uncountable (Cannot count)} \\
\hline
Can use a/an & Cannot use a/an \\
\hline
Have plural form & No plural form \\
\hline
apple $\rightarrow$ apples & water (NOT: waters) \\
book $\rightarrow$ books & information (NOT: informations) \\
car $\rightarrow$ cars & money (NOT: moneys) \\
\hline
Use: many, few, a few & Use: much, little, a little \\
\hline
\textbf{Examples:} & \textbf{Examples:} \\
an apple, two apples & water, some water \\
a book, many books & information, much information \\
\hline
\end{tabular}
\caption{Countable vs. Uncountable nouns}
\end{table}

\begin{vocabbox}[title=Common Uncountable Nouns]
\begin{itemize}
    \item \textbf{Liquids:} water, milk, coffee, tea, juice, oil
    \item \textbf{Materials:} wood, paper, glass, plastic, gold
    \item \textbf{Food:} bread, rice, cheese, meat, butter, sugar
    \item \textbf{Abstract:} information, advice, knowledge, work, homework
    \item \textbf{Other:} money, furniture, luggage, weather, news
\end{itemize}

\textbf{To make them countable, use:}
\begin{itemize}
    \item a glass of water, two cups of coffee
    \item a piece of information, some pieces of advice
    \item a loaf of bread, three slices of cheese
\end{itemize}
\end{vocabbox}

\section{Basic Word Order: SVO}

English follows a strict word order: \textbf{Subject + Verb + Object}

\begin{grammarbox}[title=Word Order Rules]
\textbf{Basic Structure:} S + V + O + Place + Time

\textbf{Examples:}
\begin{itemize}
    \item I \textbf{(S)} study \textbf{(V)} English \textbf{(O)} at home \textbf{(Place)} every day \textbf{(Time)}.
    \item She \textbf{(S)} reads \textbf{(V)} books \textbf{(O)} in the library \textbf{(Place)} on weekends \textbf{(Time)}.
\end{itemize}

\textbf{Common Errors:}
\begin{itemize}
    \item \textcolor{red}{I every day study English.} \textbf{X}
    \item \textcolor{green!60!black}{I study English every day.} \checkmark
\end{itemize}
\end{grammarbox}

\section{Key Takeaways}
\begin{itemize}
    \item Use \textbf{a/an} for singular countable nouns (first mention or general)
    \item Use \textbf{the} for specific things both people know about
    \item Don't use articles with: general plurals, uncountable nouns (general), most countries, languages
    \item \textbf{Countable} nouns have plurals; \textbf{uncountable} don't
    \item English word order: Subject + Verb + Object + Place + Time
    \item Never say "the my book" - use possessives alone
\end{itemize}
