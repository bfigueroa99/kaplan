\section{Consonant Pronunciation}

\begin{vocabbox}[title=Tricky Consonant Sounds]
\begin{itemize}
    \item \textbf{Th /θ/ and /ð/:} Think, this, that
    \item \textbf{Ch /tʃ/:} Choose, change, church
    \item \textbf{Sh /ʃ/:} Show, sure, wish
    \item \textbf{Ph /f/:} Phone, photo (sounds like F)
    \item \textbf{Gh:} Often silent - right, light, thought
    \item \textbf{Ng /ŋ/:} sing, ring, thing (not "ng" as in "hang")
\end{itemize}
\end{vocabbox}

\section{Stress and Intonation}

\begin{grammarbox}[title=Word Stress]
\textbf{English is a stress-timed language.} Some syllables are stressed (louder, longer), others unstressed.

\textbf{Common stress patterns:}
\begin{itemize}
    \item 2-syllable words: Stress first syllable usually
    \begin{itemize}
        \item \textbf{PREsent} (noun), \textbf{TAble}, \textbf{SIster}
    \end{itemize}
    \item 3-syllable words: Often first or second syllable
    \begin{itemize}
        \item \textbf{FAMily}, \textbf{imPORtant}, \textbf{PERfect}
    \end{itemize}
    \item Verbs vs. Nouns: Sometimes stress differs
    \begin{itemize}
        \item \textbf{PREsent} (noun - gift), pre\textbf{SENT} (verb - give)
        \item \textbf{REcord} (noun), re\textbf{CORD} (verb)
    \end{itemize}
\end{itemize}
\end{grammarbox}

\section{Common Pronunciation Problems for Spanish Speakers}

\begin{tcolorbox}[colback=yellow!5, colframe=orange!30, title=Difficult Sounds for Spanish Speakers]

\textbf{1. /ðə/ vs /ðɪ/ (the):}
\begin{itemize}
    \item Before vowels: "thuh" /ðə/ - the apple
    \item Before consonants: "thee" /ðɪ/ - the book
\end{itemize}

\textbf{2. R sound /r/:}
\begin{itemize}
    \item Spanish: Rolled or tapped R
    \item English: Back of mouth, not rolled - \textbf{right}, \textbf{red}
\end{itemize}

\textbf{3. V sound /v/:}
\begin{itemize}
    \item Not pronounced as Spanish B
    \item Use upper teeth and lower lip: \textbf{very}, \textbf{love}
\end{itemize}

\textbf{4. TH /θ/ and /ð/:}
\begin{itemize}
    \item Doesn't exist in Spanish
    \item Place tongue between teeth: \textbf{think}, \textbf{this}
\end{itemize}

\textbf{5. Unstressed vowels reduce to /ə/ (schwa):}
\begin{itemize}
    \item "a" in about = ə
    \item Spanish doesn't reduce vowels like English does
\end{itemize}

\end{tcolorbox}

\section{Key Takeaways}

\begin{itemize}
    \item English has stress-timed rhythm, unlike Spanish
    \item Stress on wrong syllable can change meaning
    \item Many English words have silent letters
    \item Vowels reduce to /ə/ in unstressed syllables
    \item Spanish speakers often struggle with R, V, and TH sounds
    \item Proper pronunciation improves comprehension and fluency
\end{itemize}
